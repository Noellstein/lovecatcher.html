<script>
const canvas = document.getElementById("gameCanvas");
const ctx = canvas.getContext("2d");
const restartBtn = document.getElementById("restartBtn");
const bgm = document.getElementById("bgm");

let player, hearts, score, misses, level, speedIncrease, gameOver;
let glow = 0;

// Reset / mulai game
function startGame() {
    player = { x: 160, y: 540, size: 40 };
    hearts = [];
    score = 0;
    misses = 0;
    level = 1;
    speedIncrease = 0;
    gameOver = false;
    restartBtn.style.display = "none";
    bgm.play();
}

// Hati jatuh tiap 1 detik
setInterval(() => {
    if (!gameOver) {
        hearts.push({
            x: Math.random() * 320 + 20,
            y: -20,
            size: 20,
            speed: 3 + speedIncrease
        });
    }
}, 1000);

// Input sentuh kiri/kanan
canvas.addEventListener("touchstart", (e) => {
    let x = e.touches[0].clientX - canvas.getBoundingClientRect().left;
    if (x < canvas.width / 2) player.x -= 50;
    else player.x += 50;

    if (player.x < 0) player.x = 0;
    if (player.x > canvas.width - player.size) player.x = canvas.width - player.size;
});

// Loop game
function update() {
    if (gameOver) return;

    ctx.clearRect(0, 0, canvas.width, canvas.height);

    // Player: 2 hati (romantis)
    drawDoubleHeart(player.x, player.y, player.size);

    glow += 0.1;

    for (let i = 0; i < hearts.length; i++) {
        let h = hearts[i];
        h.y += h.speed;

        // animasi hati otomatis berkedip
        let pulse = 2 * Math.sin(glow) + h.size;

        drawHeart(h.x, h.y, pulse);

        // tertangkap
        if (Math.abs(player.x - h.x) < 40 && Math.abs(player.y - h.y) < 40) {
            score++;
            hearts.splice(i, 1);
            i--;

            if (score % 10 === 0) levelUp();
        }

        // jatuh missed
        else if (h.y > 620) {
            misses++;
            hearts.splice(i, 1);
            i--;
            if (misses >= 3) endGame();
        }
    }

    ctx.fillStyle = "#d6336c";
    ctx.font = "20px Arial";
    ctx.fillText("Score: " + score, 10, 25);
    ctx.fillText("Miss: " + misses, 10, 50);
    ctx.fillText("Level: " + level, 260, 25);

    requestAnimationFrame(update);
}

// gambar hati
function drawHeart(x, y, size) {
    ctx.fillStyle = "#ff4d6d";
    ctx.beginPath();
    ctx.moveTo(x, y);
    ctx.bezierCurveTo(x - size, y - size, x - size, y + size, x, y + size);
    ctx.bezierCurveTo(x + size, y + size, x + size, y - size, x, y);
    ctx.fill();
}

// gambar karakter player (double heart)
function drawDoubleHeart(x, y, size) {
    drawHeart(x - 15, y, size * 0.9);
    drawHeart(x + 15, y, size * 0.9);
}

function levelUp() {
    level++;
    speedIncrease += 0.5;

    ctx.fillStyle = "#ff1f6a";
    ctx.font = "22px Arial";

    let romanticWords = [
        "Kamu hebat ‚ù§Ô∏è",
        "Cinta makin kuat üíû",
        "Level naik, cinta juga naik üíò",
        "Semakin dekat di hati üíó",
        "Hebat! Kamu memang spesial üíü"
    ];

    let msg = romanticWords[Math.floor(Math.random() * romanticWords.length)];

    ctx.fillText(msg, 60, 300);
}

function endGame() {
    gameOver = true;
    bgm.pause();

    ctx.fillStyle = "#c2185b";
    ctx.font = "32px Arial";
    ctx.fillText("Game Over üíî", 100, 260);

    ctx.font = "22px Arial";
    ctx.fillText("Score: " + score, 130, 300);
    ctx.fillText("Cinta itu seperti game...", 60, 350);
    ctx.fillText("kalau dijaga pasti menang ‚ù§Ô∏è", 40, 380);

    restartBtn.style.display = "inline-block";
}

restartBtn.onclick = () => {
    startGame();
    update();
};

startGame();
update();
</script>
